// lib/features/transactions/domain/scheduled_transaction.dart
@freezed
class ScheduledTransaction with _$ScheduledTransaction {
  const factory ScheduledTransaction({
    required String id,
    required String accountId,
    required Money amount,             // signed milliunits
    String? payeeId,
    String? categoryId,
    String? memo,
    required ScheduleFrequency frequency,
    required DateTime nextDate,        // next occurrence to materialize
  }) = _ScheduledTransaction;
}

// lib/features/transactions/domain/scheduled_transaction_repository.dart
abstract interface class ScheduledTransactionRepository {
  /// All schedules for accounts in [budgetId], re-emitted on change.
  Stream<List<ScheduledTransaction>> watchAll(String budgetId);
  /// Schedules whose `nextDate` is on or before [asOf] — the materialization
  /// queue (M4-T20).
  Future<List<ScheduledTransaction>> due(String budgetId, DateTime asOf);
  Future<ScheduledTransaction?> findById(String id);
  Future<void> save(ScheduledTransaction schedule);
  Future<void> delete(String id);
}

// lib/data/daos/scheduled_transactions_dao.dart
@DriftAccessor(tables: [ScheduledTransactions, Accounts])
class ScheduledTransactionsDao extends DatabaseAccessor<AppDatabase>

  ScheduledTransactionsDao(super.db);
  Stream<List<ScheduledTransactionRow>> watchByBudget(String budgetId);

  Future<ScheduledTransactionRow?> findById(String id);
  Future<void> upsert(ScheduledTransactionsCompanion schedule);
  Future<void> deleteById(String id);

class ScheduledTransactionRepositoryImpl

  ScheduledTransactionRepositoryImpl(AppDatabase db);

## Implementation notes
- `Accounts` is in the `@DriftAccessor` list only so the DAO can join
  `scheduled_transactions.accountId` to `accounts.id` and filter by
  `accounts.budgetId` — `scheduled_transactions` itself has no `budget_id`.
- `nextDate` is a `DateTime` on the entity, `YYYY-MM-DD` text in the column.
  Map it date-only, same as `transactions.date` in M1-T21. `dueByBudget`
  compares the text column against the `asOf` date string —
  `YYYY-MM-DD` strings sort and compare correctly.
- `amount` is signed `Money`.
- `scheduled_transactions` has `created_at` / `updated_at`; `save` preserves
  `createdAt` across updates (same pattern as M1-T14).
- `delete` is a hard delete — schedules are not soft-deleted (`docs/03`).
- The impl lives in `data/`; the domain entity and interface stay Drift-free.
